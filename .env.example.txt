# ==========================================
# ⭐ Group 1 — 個股策略參數（V1.1 格式）
# ==========================================
# 每支股票一個 PC_<代號> 變數，JSON 格式。
#
# 通用參數：
#   strategy      策略名稱（bollinger / vwap / ma_cross / breakout / keep_wait）
#   alloc         資金配置百分比（相對於 TOTAL_CAPITAL）
#   position_amount   單筆買進金額（元，bollinger/vwap/ma_cross 適用）
#   monthly_budget    每月預算上限（0 = 不限制）
#
# 策略特定參數 (對應 strategy 函式參數名稱)：
#   bollinger  | window, std_dev, rsi_period
#   vwap       | sigma_mult, rsi_period
#   ma_cross   | fast_period, slow_period, atr_threshold
#   breakout   | lookback, atr_period, atr_threshold
#
# 金字塔加碼（限 bollinger）：
#   pyramid_enabled, pyramid_tier1_shares, pyramid_tier2_shares,
#   pyramid_tier3_shares, pyramid_tier2_drop, pyramid_tier3_drop
#
# DCA 低接（限 keep_wait）：
#   initial_shares, add_drop_pct, add_shares, max_additions,
#   tp_trigger_pct, tp_sell_ratio, cooldown_days
#
# 突破策略（限 breakout）：
#   buy_shares, sell_shares
# ==========================================

# Group 2 — 法人抬轎動能策略（獨立資金池）
# ==========================================
INST_MOM_CAPITAL=500000             # 法人抬轎動能 獨立本金（0=不啟用，設金額啟用）
INST_MOM_MIN_VOLUME=2000            # 最小近 5 日平均成交量（張）
INST_MOM_BUY_RATIO=0.03             # 投信+外資買超佔總成交量門檻（3%）
INST_MOM_LOOKBACK=20                # 創新高天數
INST_MOM_TOP_N=3                    # 持有標的數量
INST_MOM_STOP_LOSS=0.07             # 硬性停損 -7%
INST_MOM_TRAILING_PERIOD=10         # 移動停利 MA10
INST_MOM_EXCLUDE_ETF=true           # 排除 ETF（true/false）
# ── 魚過濾（兩階段篩選：法人低吃 → 動能進場）─────────
INST_MOM_FISH_FILTER=true           # 啟用魚過濾（true/false）
INST_MOM_FISH_DAYS=60               # 魚過濾回溯天數
INST_MOM_FISH_MIN_SCORE=4.0         # 魚過濾最低分數門檻
# ── 獲利滾入（贖回時利潤滾入資金池）─────────────────
INST_MOM_PROFIT_ROLL_MONTHS=0       # M=0: 每筆獲利都滾入；M>0: 每 M 個月滾一次
INST_MOM_PROFIT_ROLL_PCT=1.0        # 滾入比例（0.0~1.0，1.0=100%）
# ── 停損禁入（虧損賣出後 N 天禁再進場）──────────────
INST_MOM_LOSER_BAN_DAYS=0           # 停損禁入天數（0=不啟用）
